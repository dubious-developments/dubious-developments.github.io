    MxGraph.prototype.addBlock = function (graph, b) {
        var _this = this;
        var model = graph.getModel();
        // Gets the default parent for inserting new cells. This
        // is normally the first child of the root (ie. layer 0).
        var parent = graph.getDefaultParent();
        var v1 = model.cloneCell(this.state.nameToStandardCellDict.getValue('block'));
        // Adds cells to the model in a single step
        model.beginUpdate();
        try {
            var longestname_1 = 0;
            b.traits.forEach(function (trait) {
                var temprow = model.cloneCell(_this.state.nameToStandardCellDict.getValue('row'));
                var name = trait[0] + ": " + trait[1];
                longestname_1 = Math.max(name.length, longestname_1);
                temprow.value = { name: name, trait: trait };
                v1.insert(temprow);
            });
            var edgeElements = 0;
            for (var i = 0; i < model.getChildCount(parent); i++) {
                if (!model.isEdge(model.getChildAt(parent, i))) {
                    edgeElements++;
                }
            }
            v1.value = b;
            v1.style = b.blockType;
            v1.geometry.width += longestname_1 * 4;
            graph.addCell(v1, parent);
            v1.geometry.alternateBounds =
                new mxRectangle(0, 0, v1.geometry.width, v1.geometry.height);
        }
        finally {
            // save graph into state
            // this.saveGraph(graph);
            // Updates the display
            model.endUpdate();
        }
        graph.setSelectionCell(v1);
